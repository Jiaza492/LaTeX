  P \{ X \ge 6 \} &=& 0.4232 \\
  \Rightarrow P \{ X \le 6 \} &=& 1 - 0.4232 = 0.5678

b)
\begin{eqnarray*}
  P \{ X \ge 3 \} &=& 0.8088 \\
  P \{ X \ge 9 \} &=& 0.1735 \\
  \Rightarrow   P \{ 3 \le X \le 9 \}
  &=& P \{ X \le 9 \} - P \{ X \le 3 \} \\
  &=& (1 - P \{ X \ge 9 \}) - (1 - P \{ X \ge 3 \}) \\
  &=& (1 - 0.1735) - (1 - 0.8088) \\
  &=& 0.6353
\end{eqnarray*}
